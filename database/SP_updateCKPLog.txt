BEGIN
  IF EXISTS(SELECT * FROM ckp_log_bulanan
    		WHERE user_id = 'user_id' AND month = month AND year = year) 
  THEN
    UPDATE ckp_log_bulanan,
   	  (SELECT 
            COALESCE(SUM(target_kuantitas),0) as target_kuantitas,
            COALESCE(SUM(realisasi_kuantitas),0) as realisasi_kuantitas,
            COALESCE(SUM(kualitas),0) as kualitas,
            COALESCE(SUM(kecepatan),0) as kecepatan,
            COALESCE(SUM(ketepatan),0) as ketepatan,
            COALESCE(SUM(ketuntasan),0) as ketuntasan
            FROM ckps
            WHERE user_id='user_id' AND month = month AND year = year) AS cdata 

      SET  
        target_kuantitas = cdata.target_kuantitas, 
        realisasi_kuantitas = cdata.realisasi_kuantitas, 
        kualitas = cdata.kualitas, 
        kecepatan = cdata.kecepatan, 
        ketepatan = cdata.ketepatan, 
        ketuntasan = cdata.ketuntasan 
      
      WHERE user_id = user_id AND month = month AND year = year ;
  ELSE
    INSERT INTO ckp_log_bulanan 
      (target_kuantitas, realisasi_kuantitas, kualitas, 
      kecepatan, ketepatan, ketuntasan, 
      user_id, month, year, created_by, updated_by) 

    (SELECT 
            COALESCE(SUM(target_kuantitas),0) as target_kuantitas,
            COALESCE(SUM(realisasi_kuantitas),0) as realisasi_kuantitas,
            COALESCE(SUM(kualitas),0) as kualitas,
            COALESCE(SUM(kecepatan),0) as kecepatan,
            COALESCE(SUM(ketepatan),0) as ketepatan,
            COALESCE(SUM(ketuntasan),0) as ketuntasan,
            user_id, month, year, user_id_int, user_id_int
            FROM ckps
            WHERE user_id='user_id' AND month = month AND year = year);
  END IF;
END
